
Master Interface:

	master.pd

Receiver: master
Possible Messages:

	// MOD means this parameter can be connected to the LFO or Envelope

	chorus on <float>			0 or 108
	chorus rate <float>			> 0, frequency in Hz					MOD
	chorus depth <float>		usually from 0 to 1, can be any value	MOD

	tremolo on <float> 			0 or 1
	tremolo rate <float> 		> 0, frequency in Hz					MOD
	tremolo depth <float> 		usually 0 to 1, can be any number		MOD
	tremolo shape <float> 		0 - sine, 1 - sawtooth, 2 - reverse sawtooth, 3 - square

	delay on <float> 			0 or 1
	delay feedback <float>		from 0 to 1 							MOD
	delay time <float> 			time in ms 								MOD
	delay amount <float> 		0 to 1, can be any number 				MOD

	ringmod on <float> 			0 or 1
	ringmod rate <float> 		> 0, frequency in Hz 					MOD
	ringmod depth <float> 		0 to 1 									MOD

	filter on <float> 			0 or 1
	filter frequency <float> 	> 40 and < 22000 - frequency in Hz 		MOD
	filter resonance <float> 	> 0, usually from 0 to ~20  			MOD
	filter type <symbol>		"lowpass", "bandpass", "highpass", "notch", "peak", "highshelf", "lowshelf"

	reverb on <float> 			0 or 1
	reverb size <float> 		0 to 1, size of room
	reverb damping <float> 		0 to 1, reflectiveness of the room
	reverb width <float> 		0 to 1, stereo width
	reverb wet <float> 			0 to 1, amount of signal after reverb
	reverb dry <float> 			0 to 1, amount of signal before reverb

	envelope amount <float> 	0 to 1, 0 is default and means off
	envelope adsr <list> 		"attack, decay, sustain, release" - sustain is 0 to 1, rest are time in ms (0 - 10000)
	envelope invert <float> 	0 is normal, 1 is inverted (goes in opposite direction)
	envelope dest <symbol>		possible destination is any effect parameter besides toggles ("_" instead of " "):
								"filter_frequency", "filter_resonance", "delay_feedback", "chorus_depth", "pitchbend", etc
								gain/amplitude has a separate envelope

	lfo amount <float> 			0 to 1, 0 is default and means off
	lfo rate <float> 			> 0, frequency in Hz  (0 to 50 works, also cool if it's > 50)
	lfo waveform <float> 		0 - triangle, 1 - sawtooth, 2 - reverse sawtooth, 3 - square
	lfo dest <symbol>			possible destination is any effect parameter besides toggles ("_" instead of " "):
								"filter_frequency", "filter_resonance", "delay_feedback", "gain", "pitchbend", "pan"

	gain <float> 				1 is default (no change), 0 to 1 is quieter, >1 is louder 		MOD
	volume <float> 				same as gain, but only from 0 to 1
	pan <float> 				0 to 1, 0 is left, 1 is right, .5 is equal in both speakers  	MOD
